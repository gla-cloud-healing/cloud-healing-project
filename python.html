Overview
How the healing script works
🔄
Infinite Loop
Runs forever until Ctrl+C. Checks /health every 10 seconds, 24/7, no human needed.
📊
Failure Counter
Tracks consecutive failures. Resets to 0 on success. Triggers heal at count=2.
⚡
Auto Restart
Uses subprocess to run docker restart command when threshold is reached.
📝
Full Logging
Every event (OK, DOWN, WARNING, CRITICAL, RESTART, SUCCESS) logged to alerts.log with timestamps.

Code
automation/healing_script.py — Full Code
automation/healing_script.py 
import requests
import subprocess
import time
from datetime import datetime

APP_URL     = "http://localhost:5000/health"
CONTAINER   = "my-app"
THRESHOLD   = 2   # failures before heal
CHECK_EVERY = 10  # seconds
LOG_FILE    = "automation/alerts.log"

def log(level, msg):
    ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    line = f"[{ts}] [{level}] {msg}"
    print(line)
    with open(LOG_FILE, "a") as f:
        f.write(line + "\n")

def check_health():
    try:
        r = requests.get(APP_URL, timeout=5)
        return r.status_code == 200
    except requests.exceptions.ConnectionError:
        return False

def heal_container():
    log("RESTART", "Attempting to restart container...")
    subprocess.run(["docker", "restart", CONTAINER])
    time.sleep(5)
    if check_health():
        log("SUCCESS", "Container restarted successfully!")
        log("SUCCESS", "Self-healing successful!")
    else:
        log("ERROR", "Restart failed — manual intervention needed!")

# ── MAIN LOOP ──
log("START", "Self-Healing Monitor Started!")
failure_count = 0

while True:
    if check_health():
        ts = datetime.now().strftime("%H:%M:%S")
        log("OK", f"App healthy - {ts}")
        failure_count = 0
    else:
        log("DOWN", "App is DOWN - Connection failed!")
        failure_count += 1
        log("WARNING", f"Failure count: {failure_count}")

        if failure_count >= THRESHOLD:
            log("CRITICAL",
                f"App failed {failure_count} times - Triggering self-healing!")
            heal_container()
            failure_count = 0

    time.sleep(CHECK_EVERY)
